@startuml
title High-Level Training Pipeline Overview

participant "Training Controller" as TC
participant "Batch Training Stage\n(Backpropagation & Optimization)" as TrainStage
participant "Validation & Learning Rate Control\n(Warmup / Cosine Restarts)" as ValStage
participant "Model Retention & Early Stopping\n(Checkpointing Logic)" as StopStage
participant "Final Test Evaluation\n(Performance on Unseen Data)" as TestEval

loop Repeat for each epoch until stopped
    TC -> TrainStage: Execute batch-based training\n(update model parameters)
    TrainStage --> TC: Report average training metrics

    TC -> ValStage: Validate model and adjust learning rate
    ValStage --> TC: Report validation metrics and updated LR

    TC -> StopStage: Evaluate checkpointing and early stop condition
    StopStage --> TC: Decision to continue or stop
end

TC -> TestEval: Evaluate best saved model\non test dataset
TestEval --> TC: Final performance results

@enduml
