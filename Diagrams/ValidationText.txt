@startuml
participant "Training Controller" as TC
participant "Classification Model" as M
participant "Validation Data Loader" as VDL
participant "Loss Computation Module" as Loss

loop For each validation epoch or evaluation pass
    TC -> M: Switch model into evaluation mode
    TC -> TC: Reset statistics for\nloss, accuracy, and macro F1

    loop For each batch request during validation
        TC -> VDL: Request next validation mini-batch
        VDL --> TC: Provide validation images and labels

        TC -> TC: Transfer batch to selected compute device

        TC -> M: Execute forward pass without gradient updates
        M --> TC: Predicted output scores

        TC -> Loss: Compute validation loss for this batch
        Loss --> TC: Numerical loss value

        TC -> TC: Determine predicted class and\naccumulate predictions and true labels
    end

    TC -> TC: Compute aggregated validation metrics\n(loss, accuracy, macro F1)
end
@enduml
